E000001

The ENVIRONMENT variable Isn't set correctly. Look in index.php and find the following code.

/*
 * ---------------------------------------------------------------
 * APPLICATION ENVIRONMENT
 * ---------------------------------------------------------------
 *
 * You can load different configurations depending on your
 * current environment. Setting the environment also influences
 * things like logging and error reporting.
 *
 * This can be set to anything, but default usage is:
 *
 *     development
 *     testing
 *     production
 *
 * NOTE: If you change these, also change the error_reporting() code below
 *
 */
define('ENVIRONMENT', 'development');

E000002

The constants.php file could not be found.

/*
 * ------------------------------------------------------
 *  Load the framework constants
 * ------------------------------------------------------
 */
if (defined('ENVIRONMENT') && file_exists(__DIR__ . 'library/'.ENVIRONMENT.'-constants.php')) {
    require __DIR__ . 'library/' . ENVIRONMENT . '-constants.php';
} elseif (file_exists(__DIR__ . 'library/constants.php')) {
    require __DIR__ . 'library/constants.php';
} else {
    exit('The application constants could not be found. E000002');
}
